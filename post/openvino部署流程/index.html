#include <opencv2/core.hpp>
#include <opencv2/imgproc.hpp>
#include <inference_engine.hpp>

using namespace InferenceEngine;

#pragma once
#ifdef CURVILINEARPROCESSOR_EXPORTS
#define CURVILINEARPROCESSOR_API __declspec(dllexport)
#else
#define CURVILINEARPROCESSOR_API __declspec(dllimport)
#endif

/**
 * 基类.
 */
namespace  base_processor_copy {
	class  CBaseProcessor
	{
	protected:
		CBaseProcessor() {};
	public:
		virtual ~CBaseProcessor() {};
	protected:
        // 推理请求
		InferRequest infer_request;
        // 模型路径
		std::string modal_path;
        // 设备类型
		std::string device_type;
        // onnx导出时输入和输出的名字索引
		std::string input_name;
		std::string output_name;
        
	public:
        // 基类暴露4个机构
        // 用于模型初始化，基类中提供默认实现
        CURVILINEARPROCESSOR_API virtual bool Init(const std::string model_path);
        // 前处理，纯虚函数，需在子类中重写
        CURVILINEARPROCESSOR_API virtual Blob::Ptr PreProcess(const cv::Mat& src) = 0;
        // 推理过程，通用，基类中提供默认实现
        CURVILINEARPROCESSOR_API virtual bool Forward(const cv::Mat& src);
        // 后处理，纯虚函数，需在子类中重写
        CURVILINEARPROCESSOR_API virtual bool PostProcess(const float* p) = 0;
	};
}

/*****************************************************************//**
 * \file  CNNBase.cpp
 * \brief  基类实现
 * 
 * \author sclead_dl4
 * \date   January 2023
 *********************************************************************/
#include"CNNProcessorCopy.h"
/**
 * 通用模型初始化过程.
 * 
 * \param model_path onnx模型所在路径
 * \return 
 */
bool base_processor_copy::CBaseProcessor::Init(const std::string model_path)
{
        this->modal_path = model_path;
        device_type = "CPU";
        Core ie;
        CNNNetwork network = ie.ReadNetwork(model_path);
        if (network.getOutputsInfo().size() != 1)
            throw std::logic_error("Sample supports topologies with 1 output only");
        if (network.getInputsInfo().size() != 1)
            throw std::logic_error("Sample supports topologies with 1 input only");
        InputInfo::Ptr input_info = network.getInputsInfo().begin()->second;
        input_name = network.getInputsInfo().begin()->first;
        if (network.getOutputsInfo().empty()) {
            std::cerr << "Network outputs info is empty" << std::endl;
            return EXIT_FAILURE;
        }
        DataPtr output_info = network.getOutputsInfo().begin()->second;
        output_name = network.getOutputsInfo().begin()->first;

        output_info->setPrecision(Precision::FP32);
        ExecutableNetwork executable_network = ie.LoadNetwork(network, device_type);
        infer_request = executable_network.CreateInferRequest();
    return true;
}

/**
 * 通用推理过程.
 * 
 * \param src 输入图像
 * \return 
 */
bool base_processor_copy::CBaseProcessor::Forward(const cv::Mat& src)
{
    try {
        Blob::Ptr img_blob = this->PreProcess(src);
        infer_request.SetBlob(input_name, img_blob);
        infer_request.Infer();
        const Blob::Ptr output_blob = infer_request.GetBlob(output_name);
        MemoryBlob::CPtr moutput = as<MemoryBlob>(output_blob);
        if (!moutput) {
            throw std::logic_error("We expect output to be inherited from MemoryBlob, "
                "but by fact we were not able to cast output to MemoryBlob");
        }
        auto moutputHolder = moutput->rmap();
        const float* out = moutputHolder.as<const PrecisionTrait<Precision::FP32>::value_type*>();
        // 结果在后处理函数中得到
        this->PostProcess(out);
    }
    catch (const std::exception& ex) {
        std::cerr << ex.what() << std::endl;
        return EXIT_FAILURE;
    }
    return EXIT_SUCCESS;
}

#pragma once
#include"CNNProcessorCopy.h"

class CSegmentationProcess :public base_processor_copy::CBaseProcessor
{
private:
	CSegmentationProcess() {};
private:
	cv::Mat _mDst;

public:
	CSegmentationProcess(const CSegmentationProcess&) = delete;
	CSegmentationProcess operator=(const CSegmentationProcess&) = delete;
	~CSegmentationProcess() {};

public:
    // 单例模式，ps：由于通过成员变量存储处理结果，多线程环境下无法使用
	static CSegmentationProcess* getInstance()
	{
		static CSegmentationProcess instance;
		return &instance;
	}
	cv::Mat& getDst()
	{
		return _mDst;
	}
private:
    // 重写基类方法
	Blob::Ptr PreProcess(const cv::Mat& src);
	bool PostProcess(const float* p);

private:
	void blobFromImage(cv::Mat& img, Blob::Ptr& blob);
	void SegPostProcess(const float* p);
};

 * \author zsg

 *********************************************************************/
#include"test.h"

Blob::Ptr CSegmentationProcess::PreProcess(const cv::Mat& src)
{
    cv::Mat image = src.clone();
    cv::resize(image, image, cv::Size(512, 512));
    image.convertTo(image, CV_32FC3, 1.0f / 255.0f);
    Blob::Ptr img_blob = infer_request.GetBlob(input_name); 
    blobFromImage(image, img_blob);
    return img_blob;
}

/**
 * 内存转换为连续地址.
 * 
 * \param img
 * \param blob
 */
void CSegmentationProcess::blobFromImage(cv::Mat& img, Blob::Ptr& blob) {
    int channels = 3;
    int img_h = img.rows;
    int img_w = img.cols;
    float mean_rgb[3] = { 0.485, 0.456, 0.406 };
    float std_rgb[3] = { 0.229, 0.224, 0.225 };
    InferenceEngine::MemoryBlob::Ptr mblob = InferenceEngine::as<InferenceEngine::MemoryBlob>(blob);
    auto mblobHolder = mblob->wmap();

    float* blob_data = mblobHolder.as<float*>();

    for (size_t c = 0; c < channels; c++)
    {
        for (size_t h = 0; h < img_h; h++)
        {
            for (size_t w = 0; w < img_w; w++)
            {
                blob_data[c * img_w * img_h + h * img_w + w] = ((float)img.at<cv::Vec3f>(h, w)[c] - mean_rgb[c]) / std_rgb[c];
            }
        }
    }
}

/**
 * 重写父类方法.
 *
 * \param out
 * \return
 */
bool CSegmentationProcess::PostProcess(const float* out)
{
    SegPostProcess(out);
    return true;
}

void CSegmentationProcess::SegPostProcess(const float* out)
{
    int H = 512, W = 512;
    cv::Mat result = cv::Mat::zeros(cv::Size(W, H), CV_8U);
    for (size_t h = 0; h < H; h++)
    {
        for (size_t w = 0; w < W; w++)
        {
            result.at<uchar>(h, w) = out[0 * W * H + h * W + w] > out[1 * W * H + h * W + w] ? 0 : 255;
        }
    }
    _mDst = result.clone();
}
