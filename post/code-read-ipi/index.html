function [A_hat E_hat] = winRPCA_median(I, opt)
% 参数解释：
% I-输入的待求解的尺寸为mxn的矩阵
% opt - 一个至少有四个成员的结构体
%     dw,dh：patch的宽和高
%     x_step,y_step：patch分别在x方向和y方向的滑动步长
%     论文的初始部分中opt被设置为：
%     opt.dw = 50;
%     opt.dh = 50;
%     opt.x_step = 10;
%     opt.y_step = 10;
if size(I, 3) == 3;
    I = rgb2gray(I);
end
[m n] = size(I);
A_hat = zeros(m,n);
E_hat = zeros(m,n);
C = zeros(m, n);
dw = opt.dw;
dh = opt.dh;
x_step = opt.x_step;
y_step = opt.y_step;

I1 = I ; % imfilter(I, fspecial('gaussian', 5));
D = [];
% 以x_step和y_step为滑动步长在图像上取得一系列的patch
for i = 1:y_step:m-dh+1
    for j = 1:x_step:n-dw+1
        temp = I1(i:i+dh-1, j:j+dw-1);
        D = [D, temp(:)]; % D中的每一行存储一个patch
    end
end

D = mat2gray(D); % 实现图像矩阵的归一化操作
% D = D - mean2(D);
[m1 n1] = size(D);
lambda = 1 / sqrt(max(m, n));

[A1, E1] = APG_IR(D, lambda);

AA = zeros(m, n, 100);
EE = zeros(m, n, 100);
temp = zeros(dh, dw);
temp1 = zeros(dh, dw);
index = 0;
for i = 1:y_step:m-dh+1
    for j = 1:x_step:n-dw+1
        index = 1+index;
        temp(:) = A1(:, index);
        C(i:i+dh-1, j:j+dw-1) = C(i:i+dh-1, j:j+dw-1)+1;
        temp1(:) = E1(:, index);
        for ii = i:i+dh-1
            for jj = j:j+dw-1
                AA(ii,jj, C(ii,jj)) = temp(ii-i+1, jj-j+1);
                EE(ii,jj, C(ii,jj)) = temp1(ii-i+1, jj-j+1);
            end
        end
    end
end
%     C(find(C==0)) = 1000;
for i=1:m
    for j=1:n
        if C(i,j) > 0
            A_hat(i,j) = median(AA(i,j,1:C(i,j)));
            E_hat(i,j) = median(EE(i,j,1:C(i,j)));
        end
    end
end
